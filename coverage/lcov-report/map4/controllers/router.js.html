var routes = [];
 
function addRoute(path, handler) {
 
    routes.push({
        params: {
            title: path.split(':')[1]
        },
        path: path,
        fn: handler
    });
 
}
 
function match(path) {
    var matches = routes.filter(function(route) {
        return route.path === path;
    });
 
    return matches[0];
 
}
